console.log(20==20); //উভয় দিকের ভ্যালু চেক করবে যে সমান কিনা। console.log(20=="20"); //এখানে শুধু ভ্যালুই চেক করবে ,ডাটা টাইপ চেক করবেনা। console.log(20==="20"); // এখানে ভ্যালুও চেক করবে এবং ডাটা টাইপও চেক করবে তাই এটি ফলস হবে কারণ এখানে ভ্যালু এক হলেও ডাটা টাইপ এক নয়। !=(not assignment যা ভ্যালু চেক করবে), !==(not double assignment যা ডাটা টাইপ চেক করবে)
